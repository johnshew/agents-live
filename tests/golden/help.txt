commands:
  run                      Execute an agent once.
  start                    Activate cron and watcher triggers.
  stop                     Deactivate triggers and keep configuration.
  status                   List agents and runtime state.
  logs                     Query logs and correlated event timelines.
  smoketest                Run end-to-end validation.
  doctor                   Check environment and installation readiness.
  init                     Initialize the project layout.
  upgrade                  Upgrade runtime and project skill payloads.
  migrate                  Converge persisted runtime invocations.
  heartbeat                Run or manage the host heartbeat.
  uninstall                Remove host integrations and the uv tool.
  repos                    Manage registered repositories.
  completions              Generate shell completion scripts.
  dashboard                Open the interactive control panel.

global flags:
  --json                   Machine-readable output and error envelopes.
  --repo                   Pin a path or registered repository.
  --version                Show the installed version and exit.
